## 部署 docker-register
### 创建命名空间
```sh
$ kubectl create namespace docker-registry
```
### 创建PVC
* 复用之前的 NFS 文件服务即可
```yaml
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: docker-registry-pvc
  namespace: docker-registry-space
spec:
  storageClassName: managed-nfs-storage # 此处须显式设置空字符串，否则会被设置为默认的 StorageClass
  resources:
    requests:
      storage: 5Gi
  accessModes:
  - ReadWriteMany
```

### 创建 docker-registry
```yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: docker-registry
  name: docker-registry
  namespace: docker-registry-space
spec:
  replicas: 1
  revisionHistoryLimit: 5
  selector:
    matchLabels:
      app: docker-registry
  template:
    metadata:
      labels:
        app: docker-registry
    spec:
      securityContext:
        runAsUser: 0
      containers:
        - name: docker-registry 
          image: registry:2 
          imagePullPolicy: IfNotPresent
          ports:
            - containerPort: 5000
              name: web
              protocol: TCP
          resources:
            requests:
              memory: 200Mi
              cpu: "0.1"
          terminationMessagePath: /dev/termination-log
          terminationMessagePolicy: File
          volumeMounts:
            - mountPath: /var/lib/registry/
              name: docker-registry-data
      volumes:
        - name: docker-registry-data
          persistentVolumeClaim:
            claimName: docker-registry-pvc
```

### 创建Service
* 暴露统一端口出来
```yml
apiVersion: v1
kind: Service
metadata:
  name: docker-registry-service
  namespace: docker-registry-space
spec:
  ports:
    - name: port-name
      port: 5000 
      protocol: TCP
      targetPort: 5000
  selector:
    app: docker-registry
  type: ClusterIP
```

### 配置私有仓库
```yml
$ vi /etc/docker/daemon.json
{
    "insecure-registries": ["10.108.6.48:5000"]
}

# 重启服务
[root@localhost ~]# systemctl daemon-reload
[root@localhost ~]# systemctl restart docker

# 验证运行是否正常
```

```yaml

$ docker tag apache/dolphinscheduler-worker:mysql-driver-3.1.3 10.108.6.48:5000/apache/dolphinscheduler-worker:mysql-driver-3.1.3

# push 上私服
$ docker push 10.108.6.48:5000/apache/dolphinscheduler-worker:mysql-driver-3.1.3
```
