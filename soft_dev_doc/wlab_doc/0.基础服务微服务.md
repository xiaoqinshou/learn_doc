## 前言

- 市面上的各种开发框架太多太杂，不好用，所以自己开发一个符合自己习惯的中台微服务体系，致力于一次开发终身使用。

### 数据库结构
1. 字典模块
```mermaid
erDiagram
  dict_type {
    BINARY(16) id PK "uuid_v7 唯一主键"
    varchar(64) code UK "字典类型编码"
    tinyint(1) system_built "是否系统内置"
    int sort "排序"
    tinyint(1) active "是否启用"
  }

  dict_type_i18n {
    BINARY(16) id PK "uuid_v7 唯一主键"
    BINARY(16) type_id FK "字典类型ID"
    varchar(8) lang "语言代码 zh-CN/en-US/..."
    varchar(128) name "字典类型名称"
    varchar(255) description "描述"
  }

  dict_item {
    BINARY(16) id PK "uuid_v7 唯一主键"
    BINARY(16) type_id FK "字典类型ID"
    varchar(64) code "字典项编码"
    int sort "排序"
    tinyint(1) deleted "0=正常, 1=已删除"
    tinyint(1) active "1=启用, 0=禁用"
  }

  dict_item_i18n {
    BINARY(16) id PK "uuid_v7 唯一主键"
    BINARY(16) item_id FK "字典项ID"
    varchar(8) lang "语言代码"
    varchar(128) name "字典项名称"
    varchar(255) description "描述"
  }

  dict_type ||--o{ dict_type_i18n : has
  dict_type ||--o{ dict_item : has
  dict_item ||--o{ dict_item_i18n : has
```

2. 标签模块
```mermaid
erDiagram
  tag_category {
    BINARY(16) id PK "uuid_v7 唯一主键"
    varchar(64) code UK "标签分类编码"
    tinyint(1) system_built "是否系统内置"
    int sort "排序"
  }

  tag_category_i18n {
    BINARY(16) id PK "uuid_v7 唯一主键"
    BINARY(16) category_id FK "标签分类ID"
    varchar(8) lang "语言代码"
    varchar(128) name "标签分类名称"
    varchar(255) description "描述"
  }

  tag {
    BINARY(16) id PK "uuid_v7 唯一主键"
    BINARY(16) category_id FK "标签分类ID"
    varchar(64) color "标签颜色（#RRGGBB）"
    tinyint(1) active "1=启用, 0=禁用"
    tinyint(1) deleted "0=正常, 1=已删除"
    int sort "排序"
  }

  tag_i18n {
    BINARY(16) id PK "uuid_v7 唯一主键"
    BINARY(16) tag_id FK "标签ID"
    varchar(8) lang "语言代码"
    varchar(128) name "标签名称"
    varchar(255) description "描述"
  }

  tag_relation {
    BINARY(16) id PK "uuid_v7 唯一主键"
    varchar(64) target_type "关联对象类型（user/product/org等）"
    BINARY(16) target_id "关联对象ID"
    BINARY(16) tag_id FK "标签ID"
  }

  tag_category ||--o{ tag_category_i18n : has
  tag_category ||--o{ tag : has
  tag ||--o{ tag_i18n : has
  tag ||--o{ tag_relation : assigned
```

```apidesign
API: 获取字典列表
PATH: /dict-type/list
METHOD: GET

REQUEST object
  size int required >0 <100 '分页大小'
  offset int required >0 '偏移值'
  sort asc|desc '排序方式'
  names List<string> '排序字段'

RESPONSE list
  name string '字典名称'
  type string '字典类型'
  code string '字典编码'
```
